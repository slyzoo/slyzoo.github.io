Do while loops

while loop, repeats a part of the code *while* it is true
DO while loop, always executes a part of the code once, then checks a condition
*/

do{
  printf("Enter a number above 0 : ");

    sum += number;

Nested loops


  
#include <stdio.h>
int main() {
/*
nested loops, a loop inside another loop
making rectangles
*/

  int rows;
  int columns;
  char symbol;

  printf("\n Enter number of rows : ");
  scanf("%d", &rows);

  printf("\n Enter number of columns : ");
  scanf("%d", &columns);

  scanf("%c");

  printf("\n Enter symbol : ");
  scanf("%c", &symbol);

  // i(index) = 1, if i is less than or equal to rows, increment i by 1
for (int i = 1; i <= rows; i++)
{
  // j(index 2) = 1, if j is less than or equal to columns increment j by 1
  for (int j = 1; j <= columns; j++)
  {
    printf("%c", symbol);
  }
  printf("\n");
  
}
return 0;
}

  if (i == 13)
